# Copy this file to admin/.htaccess to password protect your admin folder
# Create a htpasswd using the htpasswd utility on your server and
# set AuthUserFile to the path of the htpasswd file

# Require all requests be authenticated using the htpasswd file
AuthType Basic
AuthName "QuincyKit Crash Reports"
AuthUserFile /tmp/htpasswd
Order Deny,Allow
Deny from all
Require valid-user
satisfy any

RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^(.*)$ /index.php?q=$1 [L,QSA]
